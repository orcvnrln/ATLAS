// hooks/useScreenerWebSocket.ts

export function useScreenerWebSocket(assetIds: string[]) {
  const updateAsset = useScreenerStore((state) => state.updateAsset);
  
  useEffect(() => {
    const socket = io(WEBSOCKET_URL);
    
    // Subscribe to price updates
    socket.emit('subscribe:screener', { assetIds });
    
    socket.on('price:update', (data: PriceUpdate) => {
      updateAsset(data.assetId, {
        price: data.price,
        change24h: data.change24h,
        volume: data.volume,
        lastUpdate: Date.now(),
      });
    });
    
    return () => {
      socket.emit('unsubscribe:screener');
      socket.disconnect();
    };
  }, [assetIds]);
}

// utils/exportData.ts

export function exportToCSV(assets: Asset[]) {
  const headers = ['Symbol', 'Name', 'Price', 'Change %', 'Volume', 'Market Cap'];
  const rows = assets.map(a => [
    a.symbol,
    a.name,
    a.price,
    a.change24h,
    a.volume,
    a.marketCap,
  ]);
  
  const csv = [headers, ...rows]
    .map(row => row.join(','))
    .join('\n');
  
  downloadFile('screener-export.csv', csv, 'text/csv');
}

export function exportToJSON(assets: Asset[]) {
  const json = JSON.stringify(assets, null, 2);
  downloadFile('screener-export.json', json, 'application/json');
}

function downloadFile(filename: string, content: string, type: string) {
  const blob = new Blob([content], { type });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}
