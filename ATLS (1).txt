import { useVirtualizer } from '@tanstack/react-virtual';
import { useScreenerSort } from '../hooks/useScreenerSort';

export function ScreenerTable({ assets, loading }: ScreenerTableProps) {
  const { sortedAssets, sortBy, sortOrder, handleSort } = useScreenerSort(assets);
  const parentRef = useRef<HTMLDivElement>(null);
  
  const rowVirtualizer = useVirtualizer({
    count: sortedAssets.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 48, // row height
    overscan: 10,
  });
  
  if (loading) return <TableSkeleton />;
  
  return (
    <div ref={parentRef} className="h-full overflow-auto bg-black-950">
      {/* Header */}
      <div className="sticky top-0 z-10 bg-black-900 border-b border-gray-700">
        <div className="flex items-center h-12 px-4 text-sm font-medium text-gray-300">
          <div className="w-8">
            <Checkbox />
          </div>
          <ColumnHeader 
            label="Symbol" 
            sortKey="symbol"
            width="w-32"
          />
          <ColumnHeader 
            label="Name" 
            sortKey="name"
            width="flex-1"
          />
          <ColumnHeader 
            label="Price" 
            sortKey="price"
            width="w-28"
            align="right"
          />
          <ColumnHeader 
            label="Change %" 
            sortKey="change24h"
            width="w-28"
            align="right"
          />

            label="Volume"

            width="w-32"
            align="right"

          <ColumnHeader 
            label="Market Cap"

            width="w-32"
            align="right"
          />
          {/* More columns... */}
        </div>
      </div>
      
      {/* Virtual Rows */}
      <div
        style={{
          height: `${rowVirtualizer.getTotalSize()}px`,
          position: 'relative',
        }}
      >
        {rowVirtualizer.getVirtualItems().map((virtualRow) => {
          const asset = sortedAssets[virtualRow.index];
          return (
            <ScreenerTableRow
              key={asset.id}
              asset={asset}
              style={{
                position: 'absolute',
                top: 0,
                left: 0,
                width: '100%',
                height: `${virtualRow.size}px`,
                transform: `translateY(${virtualRow.start}px)`,
              }}
            />
          );
        })}
      </div>
    </div>
  );
}

export function ScreenerTableRow({ asset, style }: RowProps) {

        )}
        onClick={() => setShowModal(true)}
      >
        {/* Checkbox */}
        <div className="w-8">
          <Checkbox 
            checked={isSelected}
            onCheckedChange={setIsSelected}
            onClick={(e) => e.stopPropagation()}
          />

          {asset.name}
        </div>
        
        {/* Price */}

        </div>

// hooks/useScreenerWebSocket.ts

export function useScreenerWebSocket(assetIds: string[]) {
  const updateAsset = useScreenerStore((state) => state.updateAsset);
  
  useEffect(() => {
    const socket = io(WEBSOCKET_URL);
    
    // Subscribe to price updates
    socket.emit('subscribe:screener', { assetIds });
    
    socket.on('price:update', (data: PriceUpdate) => {
      updateAsset(data.assetId, {
        price: data.price,
        change24h: data.change24h,
        volume: data.volume,
        lastUpdate: Date.now(),
      });
    });
    
    return () => {
      socket.emit('unsubscribe:screener');
      socket.disconnect();
    };
  }, [assetIds]);
}

function downloadFile(filename: string, content: string, type: string) {
  const blob = new Blob([content], { type });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}
